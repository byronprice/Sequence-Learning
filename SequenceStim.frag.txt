/* Retinotopy.frag.txt */

const float twopi = 2.0*3.141592653589793;

/* Values passed from vertex shader: */
varying vec4 White;
varying vec4 Grey;
varying vec4 Black;
varying float CenterX;
varying float CenterY;
varying float Radius;
varying float spatFreq;
varying float orientation;
varying float gamma;

void main()
{
    /* Query current output texel position: */
    vec2 pos = gl_TexCoord[0].xy;

    /* Convert distance, apply shift offset: */
    float x = pos[0]-CenterX;
    float y = pos[1]-CenterY;

    float orient_vec = cos(orientation)*x+sin(orientation)*y;
    float mix_coeff = pow(0.5+0.5*sin(twopi*spatFreq*orient_vec),1.0/gamma);
    float kernel = exp(-(x*x+y*y)/(Radius*Radius/3.0));
    gl_FragColor = vec4(mix_coeff,mix_coeff,mix_coeff,kernel);


}
